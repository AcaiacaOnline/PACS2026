# LIMPEZA DE SESSÕES EXPIRADAS (Diário)
# Executa às 2:00 AM todos os dias
# Remove sessões de usuário expiradas do MongoDB
# -----------------------------------------------------------------------------
0 2 * * * cd /home/$USER/planejamento-acaiaca/backend && /home/$USER/planejamento-acaiaca/backend/venv/bin/python -c "
import asyncio
from motor.motor_asyncio import AsyncIOMotorClient
from datetime import datetime
import os

async def cleanup_sessions():
    client = AsyncIOMotorClient(os.environ.get('MONGO_URL'))
    db = client[os.environ.get('DB_NAME', 'planejamento_acaiaca')]
    result = await db.user_sessions.delete_many({'expires_at': {'\$lt': datetime.utcnow()}})
    print(f'Sessões removidas: {result.deleted_count}')
    client.close()

asyncio.run(cleanup_sessions())
"
